-- Table  game
-- id | mdate | stadium	| team1	| team2

-- Table goal
-- matchid | teamid | player | gtime |

-- Table eteam
-- id | teamname |coach

-- 1. show the matchid and player name for all goals scored by Germany. To identify German players, check for: teamid = 'GER'
SELECT matchid, player FROM goal 
WHERE teamid = 'GER';

-- 2. Show id, stadium, team1, team2 for just game 1012
SELECT id,stadium,team1,team2
FROM game
WHERE id  = 1012;

-- 3. show the player, teamid, stadium and mdate for every German goal.
SELECT gl.player, gl.teamid, gm.stadium, gm.mdate
FROM goal gl JOIN game gm
ON gm.id = gl.matchid
WHERE  gl.teamid = 'GER';

-- 4. Show the team1, team2 and player for every goal scored by a player called Mario player LIKE 'Mario%'
SELECT gm.team1, gm.team2, gl.player 
FROM goal gl JOIN game gm
ON gm.id = gl.matchid
WHERE player LIKE 'Mario%'

-- 5. Show player, teamid, coach, gtime for all goals scored in the first 10 minutes gtime<=10
SELECT g.player, g.teamid, et.coach, g.gtime 
FROM goal g JOIN eteam et
ON g.teamid = et.id
WHERE gtime <= 10;

-- 6. List the dates of the matches and the name of the team in which 'Fernando Santos' was the team1 coach.
SELECT g.mdate,et.teamname 
FROM game g JOIN eteam et
ON g.team1 = et.id
Where coach = 'Fernando Santos';

-- 7. List the player for every goal scored in a game where the stadium was 'National Stadium, Warsaw'
SELECT gl.player FROM 
goal gl JOIN game gm
ON gl.matchid = gm.id
WHERE stadium = 'National Stadium, Warsaw';

-- 8. show the name of all players who scored a goal against Germany
SELECT DISTINCT gl.player FROM game gm
JOIN goal gl
ON gm.id = gl.matchid
WHERE (gm.team1 = 'ger' OR gm.team2 = 'ger')
AND gl.teamid <> 'ger';

-- 9. Show teamname and the total number of goals scored.
SELECT et.teamname, COUNT(*) as total_goals
FROM goal g JOIN eteam et
ON g.teamid = et.id
GROUP BY teamname;

-- 10. Show the stadium and the number of goals scored in each stadium.
SELECT gm.stadium, COUNT(*) AS goal_cnt
FROM game gm JOIN goal gl
ON gm.id = gl.matchid
GROUP BY stadium;

-- 11. For every match involving 'POL', show the matchid, date and the number of goals scored.
SELECT gl.matchid, gm.mdate, COUNT(*) AS goal_cnt
FROM goal gl JOIN game gm
ON gl.matchid = gm.id
WHERE gm.team1 = 'POL' OR gm.team2 = 'POL'
GROUP BY gl.matchid, gm.mdate;
	
-- 12. For every match where 'GER' scored, show matchid, match date and the number of goals scored by 'GER'
SELECT gl.matchid, gm.mdate, COUNT(*) AS goal_cnt
FROM goal gl JOIN game gm
ON gm.id = gl.matchid
WHERE gl.teamid = 'GER' 
GROUP BY gl.matchid, gm.mdate;

-- 13. List every match with the goals scored by each team.
-- For each match, show:
-- match date (mdate)
-- team1 and the number of goals scored by team1
-- team2 and the number of goals scored by team2
-- Each goal should be counted using CASE WHEN, where:
-- a goal scored by team1 contributes 1 to score1, otherwise 0
-- a goal scored by team2 contributes 1 to score2, otherwise 0
-- Include all matches, even those where no goals were scored.
-- Sort the result by date, match id, team1, and team2.
SELECT gm.mdate, gm.team1,
SUM(CASE WHEN gl.teamid = gm.team1 THEN 1 ELSE 0 END) AS score1,
gm.team2,
SUM(CASE WHEN gl.teamid = gm.team2 THEN 1 ELSE 0 END) AS score2
FROM game gm LEFT JOIN goal gl
ON gm.id = gl.matchid
GROUP BY gm.mdate, gm.id, gm.team1, gm.team2
ORDER BY gm.mdate, gm.id, gm.team1, gm.team2;
